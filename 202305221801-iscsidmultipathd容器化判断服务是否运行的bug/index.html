#!/bin/bash

psAuxOutput=$(ps aux | grep iscsid | grep -v grep | grep -v kubectl)

if [ -z "$psAuxOutput" ]; then
  echo "ps aux do not have iscsid process"
  iscsid || true
else
  echo "have iscsid process"
  echo $psAuxOutput
fi

制作流程（以 k8s.m.daocloud.io/sig-storage/csi-provisioner:v2.2.2 为例）:

docker login registry.bingosoft.net -u xxx -p xxx
# 拉取 amd64 架构镜像
docker pull k8s.m.daocloud.io/sig-storage/csi-provisioner:v2.2.2 --platform linux/amd64
docker tag k8s.m.daocloud.io/sig-storage/csi-provisioner:v2.2.2 registry.bingosoft.net/bingokube/bcs/csi-bcs-local/sig-storage/csi-provisioner:v2.2.2-amd64
docker push registry.bingosoft.net/bingokube/bcs/csi-bcs-local/sig-storage/csi-provisioner:v2.2.2-amd64

# 拉取 arm64 架构镜像
docker pull k8s.m.daocloud.io/sig-storage/csi-provisioner:v2.2.2 --platform linux/arm64
docker tag k8s.m.daocloud.io/sig-storage/csi-provisioner:v2.2.2 registry.bingosoft.net/bingokube/bcs/csi-bcs-local/sig-storage/csi-provisioner:v2.2.2-arm64
docker push registry.bingosoft.net/bingokube/bcs/csi-bcs-local/sig-storage/csi-provisioner:v2.2.2-arm64

# 创建 manifest
docker manifest create --amend --insecure registry.bingosoft.net/bingokube/bcs/csi-bcs-local/sig-storage/csi-provisioner:v2.2.2 registry.bingosoft.net/bingokube/bcs/csi-bcs-local/sig-storage/csi-provisioner:v2.2.2-amd64 registry.bingosoft.net/bingokube/bcs/csi-bcs-local/sig-storage/csi-provisioner:v2.2.2-arm64
# 添加注释
docker manifest annotate --arch "amd64" registry.bingosoft.net/bingokube/bcs/csi-bcs-local/sig-storage/csi-provisioner:v2.2.2 registry.bingosoft.net/bingokube/bcs/csi-bcs-local/sig-storage/csi-provisioner:v2.2.2-amd64
docker manifest annotate --arch "arm64" registry.bingosoft.net/bingokube/bcs/csi-bcs-local/sig-storage/csi-provisioner:v2.2.2 registry.bingosoft.net/bingokube/bcs/csi-bcs-local/sig-storage/csi-provisioner:v2.2.2-arm64
# 推送 manifest
docker manifest push --insecure registry.bingosoft.net/bingokube/bcs/csi-bcs-local/sig-storage/csi-provisioner:v2.2.2

脚本 docker-build-multi-arch.sh 可以批量制作通用镜像

#!/bin/bash

# 发生错误立即退出
set -x
set -e

fg_red="\033[31m"
fg_green="\033[32m"
fg_yellow="\033[33m"
fg_blue="\033[34m"

set_clear="\033[0m"
set_bold="\033[1m"

success_log() {
    echo -e "${fg_green}${set_bold}$1${set_clear}"
}

error_log() {
    echo -e "${fg_red}${set_bold}$1${set_clear}"
}

warning_log() {
    echo -e "${fg_yellow}${set_bold}$1${set_clear}"
}

info_log() {
    echo -e "${fg_blue}${set_bold}$1${set_clear}"
}

# 用途：docker 构建多架构镜像
# 使用方法如：bash docker-build-multi-arch.sh k8s.m.daocloud.io/sig-storage/csi-provisioner:v2.2.2 registry.bingosoft.net/bingokube/bcs/csi-bcs-local/sig-storage/csi-provisioner:v2.2.2
#  则表示把镜像 构建成多架构镜像 registry.bingosoft.net/bingokube/bcs/csi-bcs-local/sig-storage/csi-provisioner:v2.2.2
## 登录 docker 镜像仓库
## 

# 变量定义

DOCKER_REGISTRY="registry.bingosoft.net" # 镜像仓库
DOCKER_USER="admin" # 镜像仓库用户
DOCKER_PASSWORD="Harbor12345" # 镜像仓库密码

# 校验参数
if [ $# -ne 2 ]; then
    error_log "用法: $0 <源镜像> <目标镜像>"
    error_log "示例: $0 k8s.m.daocloud.io/sig-storage/csi-provisioner:v2.2.2 registry.bingosoft.net/bingokube/bcs/csi-bcs-local/sig-storage/csi-provisioner:v2.2.2"

    # 两个镜像格式必须符合 xxx:xxx 格式
    if [[ ! $1 =~ ^: ]]; then
        error_log "源镜像格式不正确"
        exit 1
    fi
    if [[ ! $2 =~ ^: ]]; then
        error_log "目标镜像格式不正确"
        exit 1
    fi
fi

# 定义源镜像
SOURCE_IMAGE=$1
SOURCE_IMAGE_NAME=$(echo ${SOURCE_IMAGE} | cut -d':' -f1)
# 定义目标镜像
TARGET_IMAGE=$2
TARGET_IMAGE_NAME=$(echo ${TARGET_IMAGE} | cut -d':' -f1)

# 输出元数据
info_log "源镜像: ${SOURCE_IMAGE}"
info_log "源镜像名称: ${SOURCE_IMAGE_NAME}"
info_log "目标镜像: ${TARGET_IMAGE}"


# 登录 docker 镜像仓库
#docker login -u ${DOCKER_USER} -p ${DOCKER_PASSWORD} ${DOCKER_REGISTRY}
# 日志
success_log "登录 docker 镜像仓库 ${DOCKER_REGISTRY} 成功"

# 遍历检查不同架构镜像是否存在
for arch in amd64 arm64; do
    if docker images --filter=reference="${SOURCE_IMAGE}-${arch}" | grep "${SOURCE_IMAGE_NAME}"; then
        info_log "镜像 ${SOURCE_IMAGE}-${arch} 已存在"
    else
        warning_log "镜像 ${SOURCE_IMAGE}-${arch} 不存在, 尝试拉取 ${SOURCE_IMAGE}"
        docker pull ${SOURCE_IMAGE} --platform=linux/${arch}

        docker tag ${SOURCE_IMAGE} ${SOURCE_IMAGE}-${arch}
    fi

    # push
    if docker images --filter=reference="${TARGET_IMAGE}-${arch}" | grep "${TARGET_IMAGE_NAME}"; then
        info_log "镜像 ${TARGET_IMAGE}-${arch} 已存在,无需重新打tag"
    else
        docker tag ${SOURCE_IMAGE}-${arch} ${TARGET_IMAGE}-${arch}
        info_log "镜像 ${TARGET_IMAGE}-${arch} 打tag完成"
    fi
    info_log "开始推送镜像 ${TARGET_IMAGE}-${arch}"
    docker push ${TARGET_IMAGE}-${arch}
    success_log "镜像 ${TARGET_IMAGE}-${arch} 推送完成"
done

# 构建多架构镜像
docker manifest create --amend --insecure ${TARGET_IMAGE} ${TARGET_IMAGE}-amd64 ${TARGET_IMAGE}-arm64
# mark
docker manifest annotate --arch "amd64" ${TARGET_IMAGE} ${TARGET_IMAGE}-amd64
docker manifest annotate --arch "arm64" ${TARGET_IMAGE} ${TARGET_IMAGE}-arm64
# push
docker manifest push --insecure ${TARGET_IMAGE}

# 输出日志
success_log "镜像 ${TARGET_IMAGE} 构建完成"

Q&A

1. docker manifest inspect is only supported on a Docker cli with experimental cli features enabled

临时开启

export DOCKER_CLI_EXPERIMENTAL=enabled

永久开启

# 创建配置目录（如果不存在）
mkdir -p ~/.docker

# 创建或编辑配置文件
cat > ~/.docker/config.json << EOF
{
  "experimental": "enabled"
}
EOF

or

# 创建或编辑系统配置文件
cat > /etc/docker/daemon.json << EOF
{
  "insecure-registries": ["registry.bingosoft.net"],
  "experimental": true
}
EOF

# 重启 docker 服务
systemctl restart docker

入口脚本 start-build-csi-image.sh

#!/bin/bash

set -x
set -e

fg_red="\033[31m"
fg_green="\033[32m"
fg_yellow="\033[33m"
fg_blue="\033[34m"

set_clear="\033[0m"
set_bold="\033[1m"

success_log() {
    echo -e "${fg_green}$1${set_clear}"
}

error_log() {
    echo -e "${fg_red}$1${set_clear}"
}

warning_log() {
    echo -e "${fg_yellow}$1${set_clear}"
}

info_log() {
    echo -e "${fg_blue}$1${set_clear}"
}

# 变量定义
code_branch="feature/dualengine"

amd64_node="192.168.64.2"
amd64_node_codedir="/root/code/bhci-csi-driver"
arm64_node="10.211.55.45"
arm64_node_codedir="/root/code/bhci-csi-driver"

image_dir="/root/csi-tars"
date_str=$(date +%m%d)

# 参数
storage_type=$1 # local/share
tag=$2 # v1.0.0

default_image_name=bhci/hci-${storage_type}-plugin

# 打印变量、参数总览
info_log "变量总览:"
info_log "代码分支: ${code_branch}"
info_log "AMD64 节点: ${amd64_node}, 代码目录: ${amd64_node_codedir}"
info_log "ARM64 节点: ${arm64_node}, 代码目录: ${arm64_node_codedir}"

info_log "参数总览:"
info_log "存储类型: ${storage_type}"
info_log "镜像标签: ${tag}"
info_log "默认镜像名称: ${default_image_name}"

# 校验参数
if [ -z "$storage_type" ] || [ -z "$tag" ]; then
    error_log "参数缺失, 请提供 storage_type (local/share) 和 tag (e.g. v1.0.0)"
    exit 1
fi

for arch in amd64 arm64; do
    node_var="${arch}_node"
    node_var_codedir="${arch}_node_codedir"
    new_image_name=registry.bingosoft.net/bingokube/bcs/csi-bcs-${storage_type}/hci-${storage_type}-plugin:${tag}-${arch}

    # 拷贝准备脚本到节点的代码目录
    scp node-prepare.sh root@${!node_var}:${!node_var_codedir}

    # 将变量值传递给远程命令
    output=$(ssh root@${!node_var} "
    #!/bin/bash
    set -e
    set -x
    
    cd ${!node_var_codedir}

    git stash
    git checkout ${code_branch}
    git fetch
    git rebase
    git log --oneline -n3 | cat

    sh node-prepare.sh
    echo '开始构建镜像'
    make ${storage_type} NAMESPACE=kube-system IMAGE_VER=${tag}
    # 检查镜像是否存在
    if ! docker images | grep ${default_image_name} | grep ${tag} > /dev/null; then
        echo '错误: 未找到镜像 ${default_image_name}:${tag}'
        exit 1
    fi

    docker images | grep ${default_image_name} | grep ${tag}

    # 打tag
    docker tag ${default_image_name}:${tag} ${new_image_name}

    docker save -o ${image_dir}/${storage_type}-${tag}-${arch}-${date_str}.tar ${new_image_name}
    " 2>&1) || {
        error_log "构建 ${arch} 架构失败, 节点: ${!node_var}"
        error_log "错误输出: $output"
        exit 1
    }

    success_log "构建 ${arch} 架构成功, 节点: ${!node_var}"
done

# 至此 arm64 和 amd64 架构的镜像都构建完成
# 构建出如下镜像
# registry.bingosoft.net/bingokube/bcs/csi-bcs-local/hci-local-plugin   v1000.0.0-arm64      ce09b149510a        2 hours ago         143MB
# registry.bingosoft.net/bingokube/bcs/csi-bcs-local/hci-local-plugin   v1000.0.0-amd64   f672409e3898   2 hours ago     120MB
# 放在目录：/root/csi-tars/ 下

scp root@${amd64_node}:${image_dir}/${storage_type}-${tag}-amd64-${date_str}.tar ${image_dir}
scp root@${arm64_node}:${image_dir}/${storage_type}-${tag}-arm64-${date_str}.tar ${image_dir}

docker load -i ${image_dir}/${storage_type}-${tag}-amd64-${date_str}.tar
docker load -i ${image_dir}/${storage_type}-${tag}-arm64-${date_str}.tar

docker images | grep hci-${storage_type}-plugin | grep ${tag}

sh docker-build-multi-arch.sh registry.bingosoft.net/bingokube/bcs/csi-bcs-${storage_type}/hci-${storage_type}-plugin:${tag} registry.bingosoft.net/bingokube/bcs/csi-bcs-${storage_type}/hci-${storage_type}-plugin:${tag}
